### Test Pattern

```typescript
describe('ColorTokenSchema validation', () => {
  // Happy path - valid inputs pass
  it('should validate correct color token', () => {
    const valid = { hex: '#FF0000', confidence: 0.95 }
    expect(ColorTokenSchema.safeParse(valid).success).toBe(true)
  })

  // Sad path - invalid inputs fail
  it('should reject invalid hex format', () => {
    const invalid = { hex: 'not-hex', confidence: 0.95 }
    expect(ColorTokenSchema.safeParse(invalid).success).toBe(false)
  })

  // Edge cases
  it('should reject hex with invalid length', () => {
    const invalid = { hex: '#FF', confidence: 0.95 }
    expect(ColorTokenSchema.safeParse(invalid).success).toBe(false)
  })

  it('should reject confidence outside [0, 1]', () => {
    const invalid = { hex: '#FF0000', confidence: 1.5 }
    expect(ColorTokenSchema.safeParse(invalid).success).toBe(false)
  })

  // Boundary conditions
  it('should accept confidence boundaries', () => {
    expect(ColorTokenSchema.safeParse({
      hex: '#FF0000', confidence: 0
    }).success).toBe(true)
    expect(ColorTokenSchema.safeParse({
      hex: '#FF0000', confidence: 1
    }).success).toBe(true)
  })
})
```
